## 1G 데이터 전송테스트

- server
    
    ```jsx
    #define _GNU_SOURCE // recvmmsg 함수 사용을 위해 선언합니다.
    #include <stdio.h> // 표준 입출력 헤더입니다.
    #include <stdlib.h> // 시스템 종료 함수 등을 위한 헤더입니다.
    #include <string.h> // 메모리 비교 및 복사를 위한 헤더입니다.
    #include <unistd.h> // 시스템 콜 함수 헤더입니다.
    #include <arpa/inet.h> // IP 주소 변환 헤더입니다.
    #include <sys/socket.h> // 소켓 통신을 위한 헤더입니다.
    #include <sys/time.h> // 처리 시간 측정을 위한 헤더입니다.
    
    #define PORT 5005 // 수신 포트 번호입니다.
    #define PACKET_SIZE 1024 // 패킷 크기입니다.
    #define BATCH_SIZE 500 // 한 번에 수신할 배치 크기입니다.
    #define EXPECTED_TOTAL 1000000 // 수신해야 할 총 패킷 수(100만 개)입니다.
    
    int main() { // 메인 함수입니다.
        int server_fd; // 서버 소켓 파일 디스크립터입니다.
        struct sockaddr_in server_addr; // 서버 주소 구조체입니다.
    
        server_fd = socket(AF_INET, SOCK_DGRAM, 0); // UDP 소켓 생성입니다.
        if (server_fd < 0) { perror("소켓 생성 실패"); exit(EXIT_FAILURE); } // 실패 처리입니다.
    
        int nRecvBuf = 8 * 1024 * 1024; // 수신 버퍼를 8MB로 확장합니다.
        setsockopt(server_fd, SOL_SOCKET, SO_RCVBUF, (const char*)&nRecvBuf, sizeof(int)); // 커널 수신 버퍼 크기를 적용합니다.
    
        memset(&server_addr, 0, sizeof(server_addr)); // 구조체 초기화입니다.
        server_addr.sin_family = AF_INET; // IPv4 설정입니다.
        server_addr.sin_addr.s_addr = INADDR_ANY; // 모든 IP 수용입니다.
        server_addr.sin_port = htons(PORT); // 포트 할당입니다.
    
        if (bind(server_fd, (const struct sockaddr *)&server_addr, sizeof(server_addr)) < 0) { perror("바인드 실패"); exit(EXIT_FAILURE); } // 바인딩 검사입니다.
    
        struct mmsghdr msgs[BATCH_SIZE]; // 메시지 수신용 구조체 배열입니다.
        struct iovec iovecs[BATCH_SIZE]; // 데이터 포인터 벡터입니다.
        char buffers[BATCH_SIZE][PACKET_SIZE]; // 실제 수신 버퍼입니다.
    
        memset(msgs, 0, sizeof(msgs)); // 구조체 배열 초기화입니다.
        for (int i = 0; i < BATCH_SIZE; i++) { // 수신 준비 세팅입니다.
            iovecs[i].iov_base = buffers[i]; // 벡터와 버퍼를 연결합니다.
            iovecs[i].iov_len = PACKET_SIZE; // 수신 크기를 지정합니다.
            msgs[i].msg_hdr.msg_iov = &iovecs[i]; // 헤더에 벡터를 넣습니다.
            msgs[i].msg_hdr.msg_iovlen = 1; // 1개의 벡터만 사용합니다.
        } // 준비 끝입니다.
    
        printf("⚡ 손실률 측정 모드: 1,000,000개 데이터 수신 대기 중...\n"); // 대기 알림입니다.
        
        int received_count = 0, is_started = 0; // 수신 패킷 수와 상태 플래그입니다.
        struct timeval start, end; // 전체 처리 속도용 시간 변수입니다.
    
        while (1) { // 무한 수신 루프입니다.
            int retval = recvmmsg(server_fd, msgs, BATCH_SIZE, MSG_WAITFORONE, NULL); // 커널 버퍼에서 뭉텅이로 읽어옵니다.
            if (retval == -1) { perror("recvmmsg 에러"); continue; } // 에러 발생 시 재시도합니다.
    
            for (int i = 0; i < retval; i++) { // 읽어온 패킷 개수만큼 순회합니다.
                char *buf = buffers[i]; // 버퍼 포인터를 가져옵니다.
                int msg_len = msgs[i].msg_len; // 수신된 데이터 길이를 확인합니다.
                
                if (msg_len < 8) continue; // 헤더와 시퀀스 번호 길이(8바이트)보다 작으면 버립니다.
    
                if (memcmp(buf, "END", 3) == 0) { // 종료 패킷인지 확인합니다.
                    if (is_started) { // 패킷을 받던 중이었다면
                        gettimeofday(&end, NULL); // 전체 종료 시간을 찍습니다.
                        double elapsed = (end.tv_sec - start.tv_sec) + (end.tv_usec - start.tv_usec) / 1000000.0; // 소요 시간을 구합니다.
                        
                        double loss_rate = ((double)(EXPECTED_TOTAL - received_count) / EXPECTED_TOTAL) * 100.0; // 수신 개수를 기반으로 손실률을 계산합니다.
                        if (loss_rate < 0) loss_rate = 0.0; // 비정상 수치 보정입니다.
    
                        printf("\n==== 🏁 수신 완료 및 손실률 통계 ====\n"); // 결과 출력 시작입니다.
                        printf("- 총 전송된 패킷 (예상): %d 개\n", EXPECTED_TOTAL); // 총 전송량입니다.
                        printf("- 실제 수신된 패킷: %d 개\n", received_count); // 수신량입니다.
                        printf("- 데이터 손실률(Drop Rate): %.4f %%\n", loss_rate); // 정확한 손실률입니다.
                        printf("- 처리 소요 시간: %.3f 초\n", elapsed); // 경과 시간입니다.
                        printf("- 평균 수신 속도: %.0f 패킷/초\n", received_count / elapsed); // 속도입니다.
                        printf("======================================\n\n"); // 출력 끝입니다.
                        
                        is_started = 0; // 다음 벤치마크를 위해 상태를 초기화합니다.
                    } // 조건문 끝입니다.
                    continue; // 다음 패킷 검사로 넘어갑니다.
                } 
                else if (memcmp(buf, "DAT", 3) == 0) { // 정상 데이터 패킷인 경우입니다.
                    if (!is_started) { // 처음 들어온 패킷이면
                        start = end; // 초기화 처리(방어코드)입니다.
                        gettimeofday(&start, NULL); // 최초 시작 시간 기록입니다.
                        is_started = 1; // 수신 상태 진입입니다.
                        received_count = 0; // 카운트 초기화입니다.
                    } // 분기 끝입니다.
                    
                    // int current_seq; memcpy(&current_seq, buf + 4, 4); // 디버깅용: 필요한 경우 시퀀스 번호를 빼서 확인할 수 있습니다.
                    received_count++; // 지연시간 계산 없이 수신 카운트만 증가시킵니다.
                } // 데이터 판별 분기 끝입니다.
            } // 배치 처리 루프 끝입니다.
        } // 메인 루프 끝입니다.
        
        close(server_fd); return 0; // 반환합니다.
    } // main 끝입니다.
    ```
    
- client
    
    ```jsx
    #define _GNU_SOURCE // sendmmsg 함수 사용을 위해 GNU 확장을 명시합니다.
    #include <stdio.h> // 표준 입출력 함수를 위한 헤더입니다.
    #include <stdlib.h> // 표준 라이브러리 함수를 위한 헤더입니다.
    #include <string.h> // 메모리 복사 등을 위한 헤더입니다.
    #include <unistd.h> // 시스템 콜 함수를 위한 헤더입니다.
    #include <arpa/inet.h> // 네트워크 주소 변환을 위한 헤더입니다.
    #include <sys/socket.h> // 소켓 프로그래밍을 위한 헤더입니다.
    #include <sys/time.h> // 전체 소요 시간 측정을 위한 헤더입니다.
    
    #define PORT 5005 // 목적지 서버의 포트 번호입니다.
    #define TOTAL_PACKETS 1000000 // 총 전송할 패킷 수(100만 개)를 정의합니다.
    #define PACKET_SIZE 1024 // 단일 패킷의 크기(1024바이트)입니다.
    #define BATCH_SIZE 500 // 한 번의 sendmmsg로 보낼 패킷 개수입니다.
    
    int main() { // 프로그램 진입점입니다.
        int client_fd; // 소켓 파일 디스크립터 변수입니다.
        struct sockaddr_in server_addr; // 서버 주소 구조체입니다.
    
        client_fd = socket(AF_INET, SOCK_DGRAM, 0); // UDP 소켓을 생성합니다.
        if (client_fd < 0) { // 소켓 생성 실패를 확인합니다.
            perror("소켓 생성 실패"); // 에러 메시지를 출력합니다.
            exit(EXIT_FAILURE); // 비정상 종료합니다.
        } // 조건문 끝입니다.
    
        memset(&server_addr, 0, sizeof(server_addr)); // 주소 구조체 메모리를 초기화합니다.
        server_addr.sin_family = AF_INET; // IPv4 방식을 사용합니다.
        server_addr.sin_port = htons(PORT); // 포트를 네트워크 바이트 순서로 변환합니다.
        server_addr.sin_addr.s_addr = inet_addr("70.12.112.112"); // 서버 IP 주소를 입력합니다.
    
        struct mmsghdr msgs[BATCH_SIZE]; // 다중 메시지 헤더 배열입니다.
        struct iovec iovecs[BATCH_SIZE]; // 데이터 포인터를 담을 I/O 벡터 배열입니다.
        char buffers[BATCH_SIZE][PACKET_SIZE]; // 실제 데이터를 담을 버퍼 배열입니다.
    
        memset(msgs, 0, sizeof(msgs)); // 헤더 배열을 초기화합니다.
        memset(buffers, 'A', sizeof(buffers)); // 버퍼를 더미 데이터('A')로 채웁니다.
    
        for (int i = 0; i < BATCH_SIZE; i++) { // 구조체 배열을 설정하는 루프입니다.
            iovecs[i].iov_base = buffers[i]; // 각 벡터가 버퍼를 가리키게 합니다.
            iovecs[i].iov_len = PACKET_SIZE; // 전송할 길이를 지정합니다.
            msgs[i].msg_hdr.msg_iov = &iovecs[i]; // 헤더에 벡터를 연결합니다.
            msgs[i].msg_hdr.msg_iovlen = 1; // 사용하는 벡터 개수를 1개로 명시합니다.
            msgs[i].msg_hdr.msg_name = &server_addr; // 목적지 주소를 설정합니다.
            msgs[i].msg_hdr.msg_namelen = sizeof(server_addr); // 목적지 주소 길이를 설정합니다.
        } // 초기화 루프 끝입니다.
    
        printf("🚀 %d개의 패킷을 전송합니다. (타임스탬프 제거, 시퀀스 번호만 삽입)\n", TOTAL_PACKETS); // 시작 알림입니다.
    
        struct timeval start, end; // 전체 소요 시간 측정용 변수입니다.
        gettimeofday(&start, NULL); // 전체 시작 시간을 기록합니다.
    
        int sent_total = 0; // 누적 전송 개수입니다.
        while (sent_total < TOTAL_PACKETS) { // 100만 개를 다 보낼 때까지 반복합니다.
            int to_send = (TOTAL_PACKETS - sent_total) < BATCH_SIZE ? (TOTAL_PACKETS - sent_total) : BATCH_SIZE; // 이번에 보낼 개수를 계산합니다.
    
            for (int i = 0; i < to_send; i++) { // 버퍼에 데이터를 채우는 루프입니다.
                int current_seq = sent_total + i; // 현재 패킷의 시퀀스 번호를 계산합니다.
                memcpy(buffers[i], "DAT\0", 4); // 데이터 패킷임을 알리는 헤더를 넣습니다.
                memcpy(buffers[i] + 4, &current_seq, 4); // 4바이트 크기의 시퀀스 번호를 삽입합니다. (타임스탬프 제외됨)
            } // 데이터 삽입 끝입니다.
    
            int retval = sendmmsg(client_fd, msgs, to_send, 0); // 커널에 일괄 전송을 요청합니다.
            if (retval == -1) { // 에러 발생 여부를 확인합니다.
                perror("sendmmsg 에러"); // 에러를 출력합니다.
                break; // 루프를 종료합니다.
            } // 조건문 끝입니다.
            usleep(500); // 1ms 간격을 둡니다. (네트워크 부하 완화를 위해)
            sent_total += retval; // 성공적으로 보낸 개수만큼 누적합니다.
        } // 전체 전송 루프 끝입니다.
    
        char end_buf[PACKET_SIZE]; // 종료 패킷용 버퍼입니다.
        memset(end_buf, 'A', PACKET_SIZE); // 빈 공간을 채웁니다.
        memcpy(end_buf, "END\0", 4); // 종료 식별자를 넣습니다.
        for(int i = 0; i < 10; i++) { // UDP 특성상 유실을 대비해 10번 보냅니다.
            sendto(client_fd, end_buf, PACKET_SIZE, 0, (const struct sockaddr *)&server_addr, sizeof(server_addr)); // 종료 패킷 전송합니다.
            usleep(5000); // 간격을 둡니다.
        } // 종료 패킷 루프 끝입니다.
    
        gettimeofday(&end, NULL); // 전체 완료 시간을 기록합니다.
        double elapsed = (end.tv_sec - start.tv_sec) + (end.tv_usec - start.tv_usec) / 1000000.0; // 소요 시간을 초 단위로 구합니다.
        
        printf("- 전송 소요 시간: %.3f 초\n", elapsed); // 경과 시간을 출력합니다.
        printf("- 발송 속도: %.0f 패킷/초\n", TOTAL_PACKETS / elapsed); // 초당 전송률을 출력합니다.
    
        close(client_fd); // 소켓을 반환합니다.
        return 0; // 프로그램 종료입니다.
    } // main 함수 끝입니다.
    ```
