# 프론트엔드 필요한 데이터 필드 목록

생성자: 재국

```markdown
erDiagram
    %% 1. 물류 아이템 (메인 정보)
    LogisticsItem {
        VARCHAR(50) tracking_number PK "운송장 번호 (고유값)"
        VARCHAR(20) destination "목표 지역 (서울/부산/대구/광주/대전)"
        ENUM status "상태 (준비/이동/완료)"
        DATETIME created_at "데이터 생성 시간"
        DATETIME completed_at "분류 완료 시간"
        DATETIME updated_at "최종 수정 시간"
    }

    %% 2. 스캔 로그 (이력 정보)
    ScanLog {
        INTEGER id PK "로그 고유 ID (Auto Inc)"
        VARCHAR(50) tracking_number FK "운송장 번호 (참조)"
        VARCHAR(20) camera_id "촬영한 카메라 ID"
        VARCHAR(20) detected_destination "AI가 인식한 지역명"
        FLOAT confidence_score "AI 신뢰도 (%)"
        DATETIME scanned_at "스캔된 시간"
    }

    %% 3. 장치 상태 (독립 테이블) - 필드 추가됨
    DeviceStatus {
        VARCHAR(50) device_id PK "기기 ID (ROBOT_01 등)"
        INTEGER battery_level "배터리 잔량 (0~100)"
        BOOLEAN is_connected "연결 상태 (True/False)"
        DATETIME last_heartbeat "마지막 통신 시간"
        FLOAT cpu_temperature "CPU 온도"
        VARCHAR(50) location "위치 정보 (좌표 또는 구역명)"
        ENUM operation_status "동작 상태 (적재/운송/정지)"
    }

    %% 관계 정의
    LogisticsItem ||--o{ ScanLog : "1개의 화물은 여러 번 스캔될 수 있음"
```

- 실시간 카메라 여러개 만들어 놓고 하나만 작동하는걸로 연동시키고 누르면 각 차량 정보 알 수 있게 만들어야 함
